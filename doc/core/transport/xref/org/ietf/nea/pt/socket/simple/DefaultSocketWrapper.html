66  /**
67   * Wrapper to encapsulate a socket and mask its functions
68   * for the transmission and receipt of TNC messages thru the
69   * contained socket and for the management of the contained socket.
70   */
71  public class DefaultSocketWrapper implements SocketWrapper {
72      private static final Logger LOGGER = LoggerFactory
73              .getLogger(DefaultSocketWrapper.class);
74  
75      private final Socket socket;
76  
77      private final TransportWriter<TransportMessage> writer;
78      private final TransportReader<TransportMessage> reader;
79      
80      private final long maxMessageLength;
81  
82      private OutputStream out;
83      private InputStream in;
84  
85      private long messageIdentifier;
86  
87      private long rxCounter;
88      private long txCounter;
89  
90      /**
91       * Creates a wrapper around the given socket using the provided attributes.
92       *
93       * @param maxMessageLength the maximum supported message length of
94       * the transport connection
95       * @param socket the socket to wrap
96       * @param writer the writer to encode a transport message
97       * as byte stream for transmission
98       * @param reader the reader to decode a byte stream
99       * into a transport message after reception
100      */
101     public DefaultSocketWrapper(final long maxMessageLength,
102             final Socket socket,
103             final TransportWriter<TransportMessage> writer,
104             final TransportReader<TransportMessage> reader) {
105         
106         this.maxMessageLength = this.maxMessageLength >= 0 ? maxMessageLength
107                 : HSBConstants.HSB_TRSPT_MAX_MESSAGE_SIZE_UNKNOWN;
108 
109         this.socket = socket;
110         this.reader = reader;
111         this.writer = writer;
112         this.messageIdentifier = 0;
113     }
114 
115 
116     @Override
117     public boolean isOpen() {
118         if (socket != null && this.in != null && this.out != null) {
119             if (!socket.isClosed() && !socket.isInputShutdown()
120                     && !socket.isOutputShutdown()) {
121                 return true;
122             }
123         }
124         return false;
125     }
126 
127     @Override
128     public void initialize() throws ConnectionException {
129         try {
130             this.in = new BufferedInputStream(socket.getInputStream());
131             this.out = new BufferedOutputStream(socket.getOutputStream());
132         } catch (IOException e) {
133             throw new ConnectionException(
134                     "Socket stream is not accessible.", e);
135         }
136         
137     }
138 
139     @Override
140     public void close() {
141         try {
142 
143             this.socket.shutdownInput();
144 
145         } catch (IOException | UnsupportedOperationException e) {
146 
147             LOGGER.warn("Socket InputStream could not be closed.", e);
148 
149         } finally {
150 
151             try {
152                 this.socket.close();
153             } catch (IOException e) {
154                 LOGGER.warn("Socket could not be closed.", e);
155             }
156 
157         }
158     }
159 
160     @Override
161     public long getIdentifier() {
162         if (this.messageIdentifier
163                 < HSBConstants.TCG_MAX_MESSAGE_IDENTIFIER) {
164             return messageIdentifier++;
165         } else {
166             this.messageIdentifier = 0;
167             return this.messageIdentifier;
168         }
169     }
170 
171     @Override
172     public long getRxCounter() {
173         return this.rxCounter;
174     }
175     
176     @Override
177     public long getTxCounter() {
178         return this.txCounter;
179     }
180      
181     @Override
182     public void resetCounters() {
183 
184         // reset rx/tx counter to measure only the integrity transports
185         long roundTrips = Math.max(this.rxCounter, this.txCounter);
186         LOGGER.debug(
187                 "Counters will be reset, current counts ( Messages received:"
188                 + rxCounter + ", Messages send:" + txCounter + ", Rounds:"
189                 + roundTrips + ")");
190         this.rxCounter = 0;
191         this.txCounter = 0;
192 
193     }
194 
195     @Override
196     public void writeToStream(final TransportMessage message)
197             throws ConnectionException, SerializationException {
198 
199         this.checkMessageLength(message.getHeader().getLength());
200 
201         if (isOpen()) {
202 
203             ByteBuffer buf = new StreamedWriteOnlyByteBuffer(this.out);
204 
205             try {
206 
207                 this.writer.write(message, buf);
208 
209                 if (LOGGER.isDebugEnabled()) {
210                     LOGGER.debug("Message bytes written: "
211                             + buf.bytesWritten());
212                 }
213 
214                 this.txCounter = (this.txCounter < Long.MAX_VALUE) ? 
215                         this.txCounter + 1 : 1;
216 
217                 if (LOGGER.isDebugEnabled()) {
218                     if (message != null) {
219                         LOGGER.debug("Message send: "
220                                 + message.toString());
221                     }
222                 }
223 
224             } catch (BufferOverflowException e) {
225                 throw new ConnectionException(
226                         "Data could not be written to stream.", e);
227             } finally {
228                 buf.clear();
229             }
230 
231         } else {
232 
233             throw new ConnectionException("The socket seems not open.");
234         }
235     }
236 
237     @Override
238     public TransportMessage readFromStream()
239             throws SerializationException, ValidationException,
240             ConnectionException {
241         
242         if (isOpen()) {
243             
244             ByteBuffer buf = new StreamedReadOnlyByteBuffer(
245                     this.in);
246             
247             try {
248 
249                 TransportMessage message = this.reader.read(buf, -1);
250                 if (LOGGER.isDebugEnabled()) {
251                     LOGGER.debug("Message bytes read: " + buf.bytesRead());
252                 }
253 
254                 this.rxCounter = (this.rxCounter < Long.MAX_VALUE) ?
255                         this.rxCounter + 1 : 1;
256 
257                 if (LOGGER.isDebugEnabled()) {
258                     if (message != null) {
259                         LOGGER.debug("Message received: "
260                             + message.toString());
261                     }
262                 }
263                 
264                 return message;
265 
266             } catch (BufferUnderflowException e) {
267                 throw new ConnectionException(
268                         "Data could not be read from stream.", e);
269             } finally {
270                 buf.clear();
271             }
272             
273         } else { 
274 
275             throw new ConnectionException("The socket seems not open.");
276         }
277     }

279     /**
280      * Checks the message length according to the {@link #maxMessageLength}
281      * class member.
282      *
283      * @param length the length of the current message
284      * @throws SerializationException if message is to large
285      */
286     private void checkMessageLength(final long length)
287             throws SerializationException {
288         if (this.maxMessageLength
289                 != HSBConstants.HSB_TRSPT_MAX_MESSAGE_SIZE_UNKNOWN
290                 && this.maxMessageLength
291                 != HSBConstants.HSB_TRSPT_MAX_MESSAGE_SIZE_UNLIMITED) {
292 
293             if (length > this.maxMessageLength) {
294 
295                 throw new SerializationException("Message is to large.", true,
296                         length, this.maxMessageLength);
297 
298             }
299         }
300     }
301 
302     @Override
303     public TransportMessage createValidationErrorMessage(
304             final ValidationException exception) throws ValidationException {
305 
306         if (exception.getReasons() != null
307                 || exception.getReasons().size() >= 0) {
308 
309             Object firstReason = exception.getReasons().get(0);
310             if (firstReason instanceof byte[]) {
311                 return PtTlsMessageFactoryIetf.createError(
312                         this.getIdentifier(), IETFConstants.IETF_PEN_VENDORID,
313                         exception.getCause().getErrorCode(),
314                         (byte[]) firstReason);
315             }
316 
317             if (firstReason instanceof PtTlsMessageHeader) {
318                 return PtTlsMessageFactoryIetf.createError(
319                         this.getIdentifier(), IETFConstants.IETF_PEN_VENDORID,
320                         exception.getCause().getErrorCode(),
321                         (PtTlsMessageHeader) firstReason);
322             }
323 
324         }
325 
326         return PtTlsMessageFactoryIetf.createError(this.getIdentifier(),
327                 IETFConstants.IETF_PEN_VENDORID,
328                 exception.getCause().getErrorCode(),
329                 new byte[0]);
330 
331     }
