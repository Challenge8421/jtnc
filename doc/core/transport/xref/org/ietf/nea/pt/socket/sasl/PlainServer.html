45  import javax.security.auth.callback.Callback;
46  import javax.security.auth.callback.CallbackHandler;

48  import javax.security.auth.callback.PasswordCallback;

55  /**
56   * Implements the PLAIN SASL server-side mechanism. (<A
57   * HREF="http://www.ietf.org/rfc/rfc4616.txt">RFC 4616</A>)
58   * It processes a name callback for the authentication ID
59   * and a password callback for the associated password as well
60   * as an authorization callback for the given credentials. 
61   */
62  public class PlainServer implements SaslServer {
63  
64      private static final String MECH_NAME = "PLAIN";
65      private static final byte SEP = '\000'; // NUL
66      private static final int MAX_MESSAGE_LENGTH = 65536;
67      private boolean completed;
68      private CallbackHandler callbackHandler;
69      private String authorizationId;
70  
71      /**
72       * Constructs a Plain mechanism.
73       * 
74       * @param callbackHandler the callback handler to use
75       * @throws SaslException if callback handler is null
76       */
77      public PlainServer(final CallbackHandler callbackHandler)
78              throws SaslException {
79          if (callbackHandler == null) {
80              throw new SaslException(
81                      "Callback handler to get username/password required.");
82          }
83  
84          this.callbackHandler = callbackHandler;
85          this.completed = false;
86      }
87  
88      @Override
89      public String getMechanismName() {
90          return MECH_NAME;
91      }
92  
93      @Override
94      public byte[] evaluateResponse(final byte[] response)
95              throws SaslException {
96          
97          if (this.completed) {
98              throw new IllegalStateException(
99                      "PLAIN authentication already completed");
100         }
101 
102         this.completed = true;
103 
104         this.checkResponse(response);
105 
106         List<byte[]> tokens = this.tokenize(response);
107 
108         String authzid;
109         String authcid;
110         char[] password;
111         
112         if (tokens.size() > 2) {
113             authzid = new String(tokens.get(0), Charset.forName("UTF8"));
114             authcid = new String(tokens.get(1), Charset.forName("UTF8"));
115             password = new char[tokens.get(2).length];
116 
117             for (int i = 0; i < tokens.get(2).length; i++) {
118                 password[i] = (char) tokens.get(2)[i];
119             }
120 
121             // remove password temp storage
122             for (int i = 0; i < tokens.get(2).length; i++) {
123                 tokens.get(2)[i] = (byte) 0;
124             }
125             tokens.remove(2);
126 
127         } else {
128             authzid = null;
129             authcid = new String(tokens.get(0), Charset.forName("UTF8"));
130             password = new char[tokens.get(1).length];
131 
132             for (int i = 0; i < tokens.get(1).length; i++) {
133                 password[i] = (char) tokens.get(1)[i];
134             }
135 
136             // remove password temp storage
137             for (int i = 0; i < tokens.get(1).length; i++) {
138                 tokens.get(1)[i] = (byte) 0;
139             }
140             tokens.remove(1);
141         }
142 
143         this.authenticate(authzid, authcid, password);
144 
145         return null;
146     }
147

149     @Override
150     public boolean isComplete() {
151         return this.completed;
152     }
153 
154     @Override
155     public String getAuthorizationID() {
156         if (!this.completed) {
157             throw new IllegalStateException(
158                     "PLAIN authentication not completed.");
159         }
160         
161         return this.authorizationId;
162     }

164     @Override
165     public byte[] unwrap(final byte[] incoming, final int offset,
166             final int len)
167             throws SaslException {
168         if (this.completed) {
169             throw new SaslException(
170                     "PLAIN supports neither integrity nor privacy.");
171         } else {
172             throw new IllegalStateException(
173                     "PLAIN authentication not completed.");
174         }
175     }

177     @Override
178     public byte[] wrap(final byte[] outgoing, final int offset, final int len)
179             throws SaslException {
180         if (this.completed) {
181             throw new SaslException(
182                     "PLAIN supports neither integrity nor privacy.");
183         } else {
184             throw new IllegalStateException(
185                     "PLAIN authentication not completed.");
186         }
187     }

189     @Override
190     public Object getNegotiatedProperty(final String propName) {
191         if (this.completed) {
192             if (propName.equals(Sasl.QOP)) {
193                 return "auth";
194             } else {
195                 return null;
196             }
197         } else {
198             throw new IllegalStateException(
199                     "PLAIN authentication not completed");
200         }
201     }

204     public void dispose() throws SaslException {
205         // Do nothing.

207     }

209     /**
210      * Clears the password in memory using zeros.
211      * @param password the reference to the password to clear 
212      */
213     private void clearPassword(char[] password) {
214         if (password != null) {
215             // zero out password
216             for (int i = 0; i < password.length; i++) {
217                 password[i] = (byte) 0;
218             }
219             password = null;
220         }
221     }
222 
223     /**
224      * Checks if the format of the received response is valid.
225      * @param response the response in raw bytes to check
226      * @throws SaslException if message format is not valid
227      */
228     private void checkResponse(final byte[] response) throws SaslException {
229         if (response == null || response.length <= 0) {
230             throw new SaslException("PLAIN response cannot be empty.",
231                     new IllegalArgumentException("Response cannot be empty."));
232         } else {
233         
234             if (response.length >= MAX_MESSAGE_LENGTH) {
235                 throw new SaslException(
236                         "PLAIN Message to long.",
237                         new IllegalArgumentException("Message to long."
238                                 + " This implementation"
239                                 + " does support a maximum length of "
240                                 + MAX_MESSAGE_LENGTH + " byte."));
241             }
242             
243             {
244                 // check if all separators are in place an  message has
245                 // the RFC format
246                 int i = 0;
247                 int separatorCount = 0;
248                 while (i < response.length && separatorCount < 2) {
249                     if (response[i] == 0) {
250                         separatorCount++;
251                     }
252                     i++;
253                 }
254                 if (separatorCount < 2 || separatorCount > 2) {
255                     throw new SaslException(
256                             "PLAIN Message format is invalid.",
257                             new IllegalArgumentException(
258                                     "Message separator count does not fit."));
259                 }
260             }
261         }
262     }

264     /**
265      * Splits the fields of the given response into a List of field tokens.
266      * @param response the raw response containing the fields
267      * @return a list of field tokens as raw byte arrays
268      */
269     private List<byte[]> tokenize(final byte[] response) {
270         // tokenize
271         int pos = -1;
272         List<byte[]> token = new ArrayList<byte[]>(3);
273         for (int i = 0; i < response.length; i++) {
274             if (response[i] == SEP) {
275                 byte[] temp = new byte[i - (pos + 1)];
276                 System.arraycopy(response, pos + 1, temp, 0, i - (pos + 1));
277                 token.add(temp);
278                 pos = i;
279             }
280             if (i == response.length - 1) {
281                 byte[] temp = new byte[response.length - (pos + 1)];
282                 System.arraycopy(response, pos + 1, temp, 0, response.length
283                         - (pos + 1));
284                 token.add(temp);
285                 pos = response.length;
286             }
287         }
288         
289         return token;
290     }

292     /**
293      * Authenticates the given credentials using callbacks.
294      *
295      * @param authzid the given authorization ID
296      * @param authcid the given authentication ID
297      * @param password the given password
298      * @throws SaslException if authentication fails due to credential mismatch
299      * or internal error.
300      */
301     private void authenticate(final String authzid, final String authcid,
302             final char[] password) throws SaslException {
303         
304         try {
305             NameCallback ncb = new NameCallback("PLAIN authentication id: ",
306                     authcid);
307 
308             PasswordCallback pcb = new PasswordCallback("PLAIN password: ",
309                     false);
310 
311             this.callbackHandler.handle(new Callback[] { ncb, pcb });
312             String authenticationID = ncb.getName();
313             char[] expectedPassword = pcb.getPassword();
314             pcb.clearPassword();
315 
316             if (authenticationID == null
317                     || authenticationID.isEmpty()
318                     || expectedPassword == null
319                     || expectedPassword.length == 0) {
320 
321                 clearPassword(password);
322                 clearPassword(expectedPassword);
323 
324                 throw new SaslException("PLAIN authentication failed.");
325             }
326 
327             if (authenticationID.equals(authcid)
328                     && Arrays.equals(expectedPassword, password)) {
329 
330                 clearPassword(password);
331                 clearPassword(expectedPassword);
332 
333                 AuthorizeCallback acb = new AuthorizeCallback(authcid,
334                         (authzid == null || authzid.isEmpty()) ? authcid
335                                 : authzid);
336 
337                 callbackHandler.handle(new Callback[] { acb });
338 
339                 if (!acb.isAuthorized()) {
340                     throw new SaslException("PLAIN authentication failed.");
341                 }
342                 
343                 this.authorizationId = acb.getAuthorizedID();
344 
345             } else {
346                 clearPassword(password);
347                 clearPassword(expectedPassword);
348                 throw new SaslException("PLAIN authentication failed.");
349             }
350 
351         } catch (IOException e) {
352             // SaslException is of type IOException!
353             clearPassword(password);
354             if (e instanceof SaslException) {
355                 throw (SaslException) e;
356             }
357             throw new SaslException("Cannot get credential information", e);
358         } catch (UnsupportedCallbackException e) {
359             clearPassword(password);
360             throw new SaslException("Cannot get credential information", e);
361         } catch (IllegalArgumentException e) {
362             clearPassword(password);
363             throw new SaslException("Cannot get credential information", e);
364         }
365         
366     }
