33  /**
34   * Default authentication implementation for a NAR.
35   * This implementation uses SASL for the authentication.
36   * Because a SASL authentication mechanism can only be
37   * executed once a new authenticator must  be added for every connection
38   * that should be authenticated using this authenticator.
39   * The authenticator is not reusable!
40   */
41  public class DefaultAuthenticationClient implements Authenticator {
42  
43      private static final Logger LOGGER = LoggerFactory
44              .getLogger(DefaultAuthenticationClient.class);
45  
46      private static final AuthenticatorTypeEnum TYPE =
47              AuthenticatorTypeEnum.AUTH_CLIENT;
48      
49      private final SaslClientMechansims mechanisms;
50      
51      private PtTlsSaslResultEnum result;
52      
53      /**
54       * Creates a default client-side authenticator with no supported
55       * SASL mechanisms.
56       * 
57       */
58      public DefaultAuthenticationClient() {
59          this(null);
60      }
61      
62      /**
63       * Creates a default client-side authenticator with a selection
64       * of supported SASL client-side mechanisms. All added SASL
65       * authentication mechanisms must not be completed or the
66       * connection establishment will fail at a later point!
67       * 
68       * @param mechanisms the supported SASL client-side mechanisms
69       */
70      public DefaultAuthenticationClient(
71              SaslClientMechansims mechanisms) {
72          this.mechanisms = (mechanisms != null) ? mechanisms
73                  : new SaslClientMechansims(); 
74          this.result = null;
75      }
76      
77      @Override
78      public AuthenticatorTypeEnum getType() {
79          return TYPE;
80      }
81      
82      @Override
83      public void authenticate(SocketMessenger connection)
84              throws ValidationException, ConnectionException,
85              SerializationException {
86          
87          List<SaslMechanismEntry> suitableMechanisms = null;
88          try {
89              
90              suitableMechanisms =
91                      this.negotiateMechanisms(connection);
92             
93              while (suitableMechanisms != null
94                      && !suitableMechanisms.isEmpty()) {
95  
96                  LOGGER.debug("Suitable Mechanisms found: "
97                          + suitableMechanisms.toString()
98                          + "\n Start Authentication.");
99                  
100                 //TODO make better mechanism selection,
101                 // then by position in list and
102                 // mechanism state "not completed".
103                 
104                 Iterator<SaslMechanismEntry> iter =
105                         suitableMechanisms.iterator();
106                 
107                 SaslClient sc = null;
108                 while (iter.hasNext() && sc == null) {
109                     
110                     sc = this.mechanisms.getMechanism(
111                         suitableMechanisms.iterator().next().getName());
112                     if (sc.isComplete()) {
113                         sc = null;
114                     }
115                 }
116                 
117                 if (sc == null) {
118                     throw new ValidationException(
119                             "Unacceptable SASL mechanism.",
120                             new RuleException(
121                                     "The collection of mechanisms "
122                                             + mechanisms.toString()
123                                             + " is not supported.",
124                                     true,
125                                     PtTlsMessageErrorCodeEnum
126                                         .IETF_INVALID_MESSAGE.code(),
127                                     PtTlsErrorCauseEnum.SASL_NAMING_MISMATCH
128                                             .id()), 0);
129                 }
130 
131                 PtTlsSaslResultEnum localResult =
132                         this.executeAuthentication(sc, connection);
133                 
134                 switch (localResult) {
135                     case SUCCESS:
136                         // everything ok, do nothing
137                         break;
138                     default:
139                         throw new ConnectionException(
140                                 "Authentication failed with reason: "
141                            + localResult.toString(), localResult);
142                 }
143                 
144                 suitableMechanisms =
145                         this.negotiateMechanisms(connection);
146                 
147             }
148         } catch (SaslException e) {
149             
150             // if on authentication fails everything is 
151             // shut down in this implementation. 
152             // This could be changed by moving this up
153             // into the for loop and not throwing a connection
154             // exception.
155             
156             TransportMessage m = null;
157             
158             StringBuilder sb = new StringBuilder();
159             sb.append("SASL Authentication failed: ");
160             
161             if (e.getCause() != null) {
162                 result = PtTlsSaslResultEnum.MECHANISM_FAILURE;
163                 sb.append(result.toString());
164                 m = PtTlsMessageFactoryIetf
165                         .createSaslResult(connection.getIdentifier(),
166                                 PtTlsSaslResultEnum.MECHANISM_FAILURE);
167                 
168             } else {
169                 result = PtTlsSaslResultEnum.FAILURE;
170                 sb.append(result.toString());
171                 m = PtTlsMessageFactoryIetf
172                 .createSaslResult(connection.getIdentifier(), result);
173             }
174 
175             connection.writeToStream(m);
176             
177             throw new ConnectionException(sb.toString(), e, result);
178         }
179        
180         this.result = PtTlsSaslResultEnum.SUCCESS;
181         
182     }

184     @Override
185     public PtTlsSaslResultEnum getAuthenticationResult() {
186         return this.result;
187     }

189     /**
190      * Executes the client-side mechanism negotiation protocol flow.
191      * 
192      * @param wrappedSocket the message handler containing the socket
193      * for message transmission
194      * 
195      * @return a list of the names for suitable mechanisms
196      * @throws SerializationException if an error occurs at message
197      * serialization
198      * @throws ValidationException if an error occurs at message parsing
199      * @throws ConnectionException if the connection is broken
200      */
201     private List<SaslMechanismEntry> negotiateMechanisms(
202             SocketMessenger wrappedSocket)
203                     throws ValidationException,
204                     ConnectionException,
205                     SerializationException {
206         
207         Set<String> supportedMechanismNames =
208                 this.mechanisms.getAllMechansims().keySet();
209         
210         List<SaslMechanismEntry> supportedMechanisms = new ArrayList<>();
211         for (String mechanismName : supportedMechanismNames) {
212             supportedMechanisms.add(new SaslMechanismEntry(mechanismName));
213         }
214         
215         List<SaslMechanismEntry> suitableMechanism = new ArrayList<>();
216         
217         TransportMessage message = null;
218         while (message == null) {
219 
220             message = wrappedSocket.readFromStream();
221 
222             if (message != null) {
223                 if (message.getValue()
224                         instanceof PtTlsMessageValueSaslMechanisms) {
225 
226                     List<SaslMechanismEntry> mechanismEntries =
227                             ((PtTlsMessageValueSaslMechanisms)
228                             message.getValue()).getMechanisms();
229                     
230                     if (mechanismEntries != null
231                             && !mechanismEntries.isEmpty()) {
232                         
233                         List<SaslMechanismEntry> filteredList =
234                                 new ArrayList<>(supportedMechanisms);
235                                 
236                         filteredList.retainAll(mechanismEntries);
237                         
238                         if (filteredList.isEmpty()) {
239     
240                             throw new ValidationException(
241                                     "Unacceptable SASL mechanism.",
242                                     new RuleException(
243                                             "The collection of mechanisms "
244                                                     + mechanismEntries
245                                                         .toString()
246                                                     + " is not supported.",
247                                             true,
248                                             PtTlsMessageErrorCodeEnum
249                                                 .IETF_INVALID_MESSAGE.code(),
250                                             PtTlsErrorCauseEnum
251                                                 .SASL_NAMING_MISMATCH.id()),
252                                                 16,
253                                             message.getHeader());
254                         }
255                     
256                         suitableMechanism = filteredList;
257                     }
258                     
259                 } else if (message.getValue()
260                         instanceof PtTlsMessageValueError) {
261                     this.handleErrorMessage(message);
262 
263                 } else {
264 
265                     throw new ValidationException(
266                             "Unexpected message received.",
267                             new RuleException(
268                                     "Message of type "
269                                             + message.getValue().getClass()
270                                                     .getCanonicalName()
271                                             + " was not expected.",
272                                     true,
273                                     PtTlsMessageErrorCodeEnum
274                                         .IETF_INVALID_MESSAGE.code(),
275                                     PtTlsErrorCauseEnum.MESSAGE_TYPE_UNEXPECTED
276                                             .id()), 0, message.getHeader());
277                 }
278             }
279 
280             // Usually exceptions would be distinguished at the point.
281             // In this case throw everything although there are minor errors,
282             // Minor error should not happen here.
283         }
284         
285         return suitableMechanism;
286        
287     }

289     /**
290      * Executes the client-side authentication protocol flow using a chosen
291      * SASL mechanism.
292      * 
293      * @param saslClient the chosen SASL client-side
294      * mechanism to use for the authentication
295      * @param wrappedSocket the message handler containing the socket
296      * for message transmission
297      * 
298      * @return the authentication result (e.g. SUCCESS)
299      * @throws SerializationException if an error occurs at message
300      * serialization
301      * @throws ValidationException if an error occurs at message parsing
302      * @throws ConnectionException if the connection is broken
303      * @throws SaslException if SASL authentication fails
304      */
305     private PtTlsSaslResultEnum executeAuthentication(SaslClient saslClient,
306             SocketMessenger wrappedSocket)
307                     throws ValidationException,
308                     ConnectionException,
309                     SerializationException, SaslException {
310         
311         PtTlsSaslResultEnum lokalResult = null;
312         
313         TransportMessage m = null;
314         if (saslClient.hasInitialResponse()) {
315             byte[] response = saslClient.evaluateChallenge(new byte[0]);
316             m = PtTlsMessageFactoryIetf
317                     .createSaslMechanismSelection(wrappedSocket.getIdentifier(),
318                             new SaslMechanismEntry(
319                                     saslClient.getMechanismName()), response);
320         } else {
321             m = PtTlsMessageFactoryIetf
322                     .createSaslMechanismSelection(wrappedSocket.getIdentifier(),
323                             new SaslMechanismEntry(
324                                     saslClient.getMechanismName()));
325         }
326         
327         wrappedSocket.writeToStream(m);
328         
329         while (!saslClient.isComplete()) {
330             
331             TransportMessage message = null;
332             while (message == null) {
333     
334                 message = wrappedSocket.readFromStream();
335     
336                 if (message != null) {
337                     if (message.getValue()
338                             instanceof PtTlsMessageValueSaslAuthenticationData)
339                     {
340     
341                         byte[] serverChallenge =
342                                 ((PtTlsMessageValueSaslAuthenticationData)
343                                         message.getValue())
344                                         .getAuthenticationData();
345                         
346                         byte[] response = saslClient
347                                 .evaluateChallenge(serverChallenge);
348                         
349                         message = PtTlsMessageFactoryIetf
350                                 .createSaslAuthenticationData(
351                                         wrappedSocket.getIdentifier(),
352                                         response);
353                         
354                         wrappedSocket.writeToStream(m);
355     
356                     } else if (message.getValue()
357                             instanceof PtTlsMessageValueSaslResult) {
358                         
359                         PtTlsMessageValueSaslResult value =
360                                 (PtTlsMessageValueSaslResult)
361                                 message.getValue();
362                         
363                         lokalResult = value.getResult();
364                         
365                         byte[] resultData = value.getResultData();
366                         
367                         if (resultData != null && resultData.length > 0) {
368                             LOGGER.warn("SASL result contained additional data."
369                                     + " This  data is ignored by "
370                                     + " this implementation.");
371                         }
372                         
373                     } else if (message.getValue()
374                             instanceof PtTlsMessageValueError) {
375                         this.handleErrorMessage(message);
376     
377                     } else {
378                         throw new ValidationException(
379                                 "Unexpected message received.",
380                                 new RuleException(
381                                         "Message of type "
382                                                 + message.getValue().getClass()
383                                                         .getCanonicalName()
384                                                 + " was not expected.",
385                                         true,
386                                         PtTlsMessageErrorCodeEnum
387                                             .IETF_INVALID_MESSAGE.code(),
388                                         PtTlsErrorCauseEnum
389                                             .MESSAGE_TYPE_UNEXPECTED.id()),
390                                             0, message.getHeader());
391                     }
392                 }
393     
394                 // Usually exceptions would be distinguished at the point.
395                 // In this case throw everything although there are minor
396                 // errors,
397                 // Minor error should not happen here.
398             }
399  
400         }
401         
402         if (lokalResult == null) {
403             TransportMessage message = null;
404             while (message == null) {
405     
406                 message = wrappedSocket.readFromStream();
407     
408                 if (message != null) {
409                     if (message.getValue()
410                             instanceof PtTlsMessageValueSaslResult) {
411                         
412                         PtTlsMessageValueSaslResult value =
413                                 (PtTlsMessageValueSaslResult)
414                                 message.getValue();
415                         
416                         lokalResult = value.getResult();
417                         
418                         byte[] resultData = value.getResultData();
419                         
420                         if (resultData != null && resultData.length > 0) {
421                             
422                             LOGGER.warn("SASL result contained additional data."
423                                     + " This  data is ignored by this"
424                                     + " implementation.");
425                         
426                         }
427     
428                     } else if (message.getValue()
429                             instanceof PtTlsMessageValueError) {
430                         this.handleErrorMessage(message);
431     
432                     } else {
433                         throw new ValidationException(
434                                 "Unexpected message received.",
435                                 new RuleException(
436                                         "Message of type "
437                                                 + message.getValue().getClass()
438                                                         .getCanonicalName()
439                                                 + " was not expected.",
440                                         true,
441                                         PtTlsMessageErrorCodeEnum
442                                             .IETF_INVALID_MESSAGE.code(),
443                                         PtTlsErrorCauseEnum
444                                             .MESSAGE_TYPE_UNEXPECTED.id()),
445                                             0, message.getHeader());
446                     }
447                 }
448     
449                 // Usually exceptions would be distinguished at the point.
450                 // In this case throw everything although there are minor
451                 // errors,
452                 // Minor error should not happen here.
453             }
454         }
455        
456         return lokalResult;
457     }

459     /**
460      * Handles an error message, if one is received.
461      * @param message the received error message
462      * @throws ConnectionException if the contained error is fatal
463      */
464     private void handleErrorMessage(TransportMessage message)
465             throws ConnectionException {
466         PtTlsMessageValueError error = (PtTlsMessageValueError) message
467                 .getValue();
468         
469         this.result = PtTlsSaslResultEnum.ABORT;
470         // An error violates the protocol flow.
471         // In this regard even a minor error must
472         // lead to a connection break up. 
473         throw new ConnectionException(
474                 "Error message received in authentication phase: "
475                         + error.toString() + "\n Connection will be closed.");
476 
477     }
