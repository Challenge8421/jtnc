37  package de.hsbremen.tc.tnc.tnccs.im.loader.simple;
38  
39  import java.io.IOException;
40  import java.lang.reflect.Constructor;
41  import java.lang.reflect.InvocationTargetException;
42  import java.net.URL;
43  import java.net.URLClassLoader;
44  import java.util.ArrayList;
45  import java.util.HashMap;
46  import java.util.HashSet;
47  import java.util.List;
48  import java.util.Map;
49  import java.util.Set;
50  
51  import org.slf4j.Logger;
52  import org.slf4j.LoggerFactory;
53  
54  import de.hsbremen.tc.tnc.tnccs.im.loader.ConfigurationEntry;
55  import de.hsbremen.tc.tnc.tnccs.im.loader.ImLoader;
56  import de.hsbremen.tc.tnc.tnccs.im.loader.exception.LoadingException;
57  import de.hsbremen.tc.tnc.util.NotNull;
58  
59  /**
60   * Default loader to load IM(C/V) according to a configuration entry.
61   * It handles the class path for the loaded IM(C/V) by managing a set
62   * of URL class loaders.
63   *
64   *
65   * @param <T> the loadable type (e.g. IMC or IMV)
66   */
67  public class DefaultImLoader<T> implements ImLoader<T> {
68  
69      static final Logger LOGGER = LoggerFactory.getLogger(DefaultImLoader.class);
70  
71      private Map<String, URLClassLoader> loaders;
72  
73      /**
74       * Creates the IM(C/V) loader.
75       */
76      public DefaultImLoader() {
77          this.loaders = new HashMap<>();
78      }
79  
80      @Override
81      public List<T> loadIms(final List<ConfigurationEntry> configs) {
82          NotNull.check("Configuration entrys cannot be null.", configs);
83          List<T> imList = new ArrayList<>();
84  
85          for (ConfigurationEntry config : configs) {
86              T im;
87              try {
88                  im = this.loadIm(config);
89                  imList.add(im);
90              } catch (LoadingException e) {
91                  LOGGER.error(
92                          "Could not load IM(C/V). IM(C/V) will be ignored.", e);
93              }
94          }
95  
96          return imList;
97      }
98  
99      @Override
100     public T loadIm(final ConfigurationEntry config) throws LoadingException {
101         NotNull.check("Configuration entry cannot be null.", config);
102         if (config instanceof DefaultConfigurationEntryImJava) {
103             T im = this
104                     .loadImFromFile((DefaultConfigurationEntryImJava) config);
105             return im;
106         }
107 
108         throw new LoadingException("Configuration entry is not of type "
109                 + DefaultConfigurationEntryImJava.class.getCanonicalName()
110                 + " and will be ignored.");
111 
112     }
113 
114     @Override
115     public void cleanUp(final Set<ConfigurationEntry> configs) {
116         if (configs != null) {
117             Set<String> paths = new HashSet<>();
118             for (ConfigurationEntry cfg : configs) {
119                 if (cfg instanceof DefaultConfigurationEntryImJava) {
120                     paths.add(((DefaultConfigurationEntryImJava) cfg).getPath()
121                             .toString());
122                 }
123             }
124 
125             Set<String> removablePaths = new HashSet<>(loaders.keySet());
126             removablePaths.removeAll(paths);
127 
128             for (String removable : removablePaths) {
129 
130                 URLClassLoader cl = this.loaders.remove(removable);
131                 try {
132                     cl.close();
133                 } catch (IOException e) {
134                     LOGGER.warn(
135                             "An IOException occured while trying to close the"
136                             + " ClassLoader. Usually this does not indicate a"
137                             + " serious problem.", e);
138                 }
139             }
140 
141         }
142     }
143 
144     /**
145      * Loads a Java based IM(C/V) as specified in the given configuration entry
146      * from file and calls the default constructor.
147      *
148      * @param config the configuration entry
149      * @return the IM(C/V)
150      * @throws LoadingException if loading fails
151      */
152     @SuppressWarnings({ "rawtypes", "unchecked" })
153     private T loadImFromFile(final DefaultConfigurationEntryImJava config)
154             throws LoadingException {
155         T im = null;
156 
157         URLClassLoader urlLoader = null;
158         if (this.loaders.containsKey(config.getPath().toString())) {
159             urlLoader = this.loaders.get(config.getPath().toString());
160         } else {
161             urlLoader = new URLClassLoader(new URL[] {config.getPath()});
162         }
163         Class clazz = null;
164         Constructor ctor = null;
165         try {
166             clazz = urlLoader.loadClass(config.getMainClassName());
167             ctor = clazz.getConstructor();
168             im = (T) ctor.newInstance();
169         } catch (ClassNotFoundException e) {
170             throw new LoadingException("Class named "
171                     + config.getMainClassName()
172                     + " was not found. IM(C/V) will be ignored.", e);
173 
174         } catch (NoSuchMethodException e) {
175             throw new LoadingException("Constructor was not found for class "
176                     + clazz.getCanonicalName() + ". IM(C/V) will be ignored.",
177                     e);
178 
179         } catch (SecurityException e) {
180             throw new LoadingException(
181                     "Constructor was not accessible due to security "
182                             + "constraints. IM(C/V) will be ignored.", e);
183 
184         } catch (InstantiationException e) {
185             throw new LoadingException("Class named "
186                     + config.getMainClassName()
187                     + " could not be instantiated."
188                     + " IM(C/V) will be ignored.", e);
189 
190         } catch (IllegalAccessException e) {
191             throw new LoadingException(
192                     "Constructor was not accessible due to accessibility"
193                             + " constraints. IM(C/V) will be ignored.", e);
194 
195         } catch (IllegalArgumentException e) {
196             throw new LoadingException("Class named "
197                     + config.getMainClassName()
198                     + " could not be instantiated."
199                     + " IM(C/V) will be ignored.", e);
200 
201         } catch (InvocationTargetException e) {
202             throw new LoadingException("Class named "
203                     + config.getMainClassName()
204                     + " could not be instantiated,"
205                     + " because the constructor has thrown"
206                     + " an exception. IM(C/V) will be ignored.", e);
207 
208         }
209 
210         return im;
211     }
