46  import java.util.Arrays;
47  import java.util.HashMap;
48  import java.util.HashSet;
49  import java.util.List;
50  import java.util.Map;
51  import java.util.Set;
52  import java.util.regex.Matcher;
53  import java.util.regex.Pattern;
54  
55  import org.slf4j.Logger;
56  import org.slf4j.LoggerFactory;
57  
58  import de.hsbremen.tc.tnc.tnccs.im.loader.ConfigurationEntry;
59  import de.hsbremen.tc.tnc.tnccs.im.loader.ConfigurationFileParser;
60  import de.hsbremen.tc.tnc.tnccs.im.loader.enums.ConfigurationLineClassifier;
61  import de.hsbremen.tc.tnc.tnccs.im.loader.enums
62  .DefaultConfigurationLineClassifierEnum;

64  /**
65   * Default parser to parse configuration lines into configuration entry objects.
66   * Currently it only supports Java based IM(C/V) configuration lines. Other line
67   * are ignored.
68   *
69   *
70   */
71  public class DefaultConfigurationFileParserImJava implements
72          ConfigurationFileParser {
73  
74      private static final Logger LOGGER = LoggerFactory
75              .getLogger(DefaultConfigurationFileParserImJava.class);
76      private static final String URL_SHEMA = "file://";
77  
78      private final ConfigurationLineClassifier lineClassifier;
79  
80      /**
81       * Creates the default parser to parse configuration lines either for IMC or
82       * for IMV.
83       *
84       * @param server true if the parser should parse configuration lines for
85       * IMV, false for IMC
86       */
87      public DefaultConfigurationFileParserImJava(final boolean server) {
88          if (server) {
89              this.lineClassifier =
90                      DefaultConfigurationLineClassifierEnum.JAVA_IMV;
91          } else {
92              this.lineClassifier =
93                      DefaultConfigurationLineClassifierEnum.JAVA_IMC;
94          }
95      }
96  
97      @Override
98      public Set<ConfigurationLineClassifier> getSupportedConfigurationLines() {
99  
100         return new HashSet<>(Arrays.asList(this.lineClassifier));
101     }
102 
103     @Override
104     public Map<ConfigurationLineClassifier, Set<ConfigurationEntry>>
105     parseConfigurationEntries(final File configFile) {
106 
107         Map<ConfigurationLineClassifier, Set<ConfigurationEntry>> entries =
108                 new HashMap<>();
109 
110         List<String> lines = this.getLines(configFile);
111         entries.put(this.lineClassifier,
112                 this.convertLinesToImConfigurationEntries(lines));
113 
114         return entries;
115 
116     }
117 
118     /**
119      * Filters the configuration file searching for parsable lines.
120      *
121      * @param configFile the file to parse
122      * @return a list of parsable lines
123      */
124     private List<String> getLines(final File configFile) {
125         List<String> lines = new ArrayList<>();
126         // Construct BufferedReader from FileReader
127         try {
128             BufferedReader br = new BufferedReader(new FileReader(configFile));
129 
130             String line = null;
131             while ((line = br.readLine()) != null) {
132                 // only get lines specifying JAVA imc/s components
133                 line = line.trim();
134                 if (line.startsWith(this.lineClassifier.linePrefix())) {
135                     lines.add(line);
136                 }
137             }
138 
139             br.close();
140         } catch (IOException e) {
141             LOGGER.error(
142                     "An error occured while trying to read the configuration "
143                             + "file. Configuration lines could not be loaded.",
144                     e);
145             // return empty line list;
146             lines = new ArrayList<>();
147         }
148 
149         return lines;
150     }
151 
152     /**
153      * Converts a list of strings to configuration entries for IM(C/V) objects.
154      *
155      * @param lines the list of strings to convert
156      * @return a set of configuration entries
157      */
158     private Set<ConfigurationEntry> convertLinesToImConfigurationEntries(
159             final List<String> lines) {
160         Set<ConfigurationEntry> imSet = new HashSet<>();
161         // if nothing to parse, return
162         if (lines == null || lines.size() <= 0) {
163             return imSet;
164         }
165         // TODO maybe a more complex pattern here
166         final String javaNaming = "([a-zA-Z_$]{1}[a-zA-Z_$0-9]*"
167                 + "(\\.[a-zA-Z_$]{1}[a-zA-Z_$0-9]*)*)";
168         final String filePath = "((?!.*//.*)(?!.*/ .*)/{1}"
169                 + "([^\\\\(){}:\\*\\?<>\\|\\\"\\'])+\\.(jar))";
170         final String imName = "\"([^\"]+)\"";
171         Pattern p = Pattern.compile("^" + this.lineClassifier.linePrefix()
172                 + " " + imName + " " + javaNaming + " " + filePath + "$");
173 
174         final int nameGroupIdx = 1;
175         final int classGroupIdx = 2;
176         // 3 is an intermediate match
177         final int pathGroupIdx = 4;
178 
179         for (String line : lines) {
180             Matcher m = p.matcher(line);
181             if (m.find()) {
182                 String name = m.group(nameGroupIdx).trim();
183                 String mainClass = m.group(classGroupIdx).trim();
184                 String path = m.group(pathGroupIdx).trim();
185 
186                 try {
187                     URL url = new URL(URL_SHEMA + path);
188                     ConfigurationEntry cfg =
189                             new DefaultConfigurationEntryImJava(
190                             name, mainClass, url);
191                     imSet.add(cfg);
192 
193                 } catch (MalformedURLException e) {
194                     LOGGER.error("MalformedURLException was thrown"
195                             + " while creating configuration entry"
196                             + " for IM(C/V )named " + name
197                             + ". The IM (C/V) will be ignored.", e);
198                 }
199             } else {
200                 LOGGER.warn("Line " + line + " does not fit into the pattern"
201                         + " for an IM(C/V) configuration line.");
202             }
203 
204         }
205 
206         return imSet;
207     }
208 }
