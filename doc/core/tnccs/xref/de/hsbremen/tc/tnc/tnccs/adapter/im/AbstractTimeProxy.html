37  package de.hsbremen.tc.tnc.tnccs.adapter.im;
38  
39  import java.util.concurrent.Callable;
40  import java.util.concurrent.CancellationException;
41  import java.util.concurrent.ExecutionException;
42  import java.util.concurrent.Executors;
43  import java.util.concurrent.Future;
44  import java.util.concurrent.ScheduledExecutorService;
45  import java.util.concurrent.TimeUnit;
46  
47  import org.slf4j.Logger;
48  import org.slf4j.LoggerFactory;
49  
50  import de.hsbremen.tc.tnc.HSBConstants;
51  import de.hsbremen.tc.tnc.exception.TncException;
52  import de.hsbremen.tc.tnc.tnccs.adapter.im.exception.TerminatedException;
53  
54  /**
55   * Generic proxy base to control the time a function
56   * call needs to finish.
57   *
58   *
59   */
60  public abstract class AbstractTimeProxy {
61  
62      private static final Logger LOGGER = LoggerFactory
63              .getLogger(AbstractTimeProxy.class);
64  
65      private final long timeout;
66  
67      /**
68       * Creates the proxy base with the given timeout for a function call.
69       *
70       * @param timeout the function timeout
71       */
72      protected AbstractTimeProxy(final long timeout) {
73          if (timeout < HSBConstants.TCG_IM_MAX_FUNCTION_RUNTIME) {
74              this.timeout = timeout;
75          } else {
76              throw new IllegalArgumentException(new StringBuilder()
77                      .append("Timeout of ").append(timeout)
78                      .append(" milliseconds is to large, ")
79                      .append("timeout must be less than one second ")
80                      .append("( < 1000 milliseconds).").toString());
81          }
82      }
83  
84      /**
85       * Executes a function contained in a callable object.
86       *
87       * @param function the callable function
88       * @throws TncException if function execution fails
89       * @throws TerminatedException if the called IM(C/V) is terminated
90       */
91      protected void execute(final Callable<Boolean> function)
92              throws TncException, TerminatedException {
93          ScheduledExecutorService exec = Executors.newScheduledThreadPool(2);
94          // Use executor to execute a command and stop it after a fixed time
95          // length.
96          final Future<?> task = exec.submit(function);
97          exec.schedule(new WatchDog(task), this.timeout, TimeUnit.MILLISECONDS);
98          // shutdown the executor after finished, timeout is 3 times the command
99          // timeout should not
100         // ever be used.
101         try {
102             // It has to be something blocking here, because the send method in
103             // state cann't send
104             // messages if the IMC has not returned from the message.
105             LOGGER.debug("Wait on the task " + task.toString() + " to finish.");
106             task.get();
107         } catch (InterruptedException e) {
108             LOGGER.debug("Thread " + Thread.currentThread().toString()
109                     + " interrupted.");
110             Thread.currentThread().interrupt();
111         } catch (ExecutionException e) {
112             Throwable t = e.getCause();
113             // Cast to TNCException and throw further;
114             if (t instanceof TncException) {
115                 throw (TncException) t;
116             } else if (t instanceof TerminatedException) {
117                 throw (TerminatedException) t;
118             }
119         } catch (CancellationException e) {
120             LOGGER.debug("Task " + task.toString()
121                     + " was cancelled, it may have taken more than "
122                     + this.timeout + " milliseconds.");
123         }
124         exec.shutdown();
125         LOGGER.debug("Task " + task.toString()
126                 + " finished and time control resources released.");
127     }
128 
129     /**
130      * Runnable, that controls the execution time of a function
131      * and stops its execution, if a timeout is reached.
132      *
133          *
134      */
135     @SuppressWarnings("rawtypes")
136     private class WatchDog implements Runnable {
137 
138         private Future task;
139 
140         /**
141          * Creates the runnable with the given task to
142          * watch out for.
143          *
144          * @param task the function task to control
145          */
146         public WatchDog(final Future task) {
147             this.task = task;
148         }
149 
150         @Override
151         public void run() {
152             if (!this.task.isDone()) {
153                 LOGGER.debug("WatchDog released to cancel method call.");
154                 this.task.cancel(true);
155             }
156         }
157     }
158 }
