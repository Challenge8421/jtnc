37  package de.hsbremen.tc.tnc.tnccs.adapter.connection;
38  
39  import org.ietf.nea.pb.message.PbMessageFactoryIetf;
40  import org.ietf.nea.pb.message.enums.PbMessageImFlagEnum;
41  import org.trustedcomputinggroup.tnc.ifimc.TNCConstants;
42  import org.trustedcomputinggroup.tnc.ifimc.TNCException;
43  
44  import de.hsbremen.tc.tnc.HSBConstants;
45  import de.hsbremen.tc.tnc.attribute.DefaultTncAttributeTypeFactory;
46  import de.hsbremen.tc.tnc.attribute.TncClientAttributeTypeEnum;
47  import de.hsbremen.tc.tnc.attribute.TncCommonAttributeTypeEnum;
48  import de.hsbremen.tc.tnc.exception.TncException;
49  import de.hsbremen.tc.tnc.message.exception.ValidationException;
50  import de.hsbremen.tc.tnc.message.tnccs.message.TnccsMessage;
51  import de.hsbremen.tc.tnc.report.enums.ImHandshakeRetryReasonEnum;
52  
53  /**
54   * IMC connection adapter adapting a simple IMC connection according to
55   * IETF/TCG specifications.
56   *
57   *
58   */
59  class ImcConnectionAdapterIetf extends AbstractImConnectionAdapter implements
60          ImcConnectionAdapter {
61  
62      private final ImcConnectionContext context;
63      private long maxMessageSize;
64  
65      /**
66       * Creates an IMC connection adapter for an IMC with the
67       * given ID and the given IMC connection context.
68       *
69       * @param primaryImcId the IMC ID
70       * @param context the connection context
71       */
72      ImcConnectionAdapterIetf(final int primaryImcId,
73              final ImcConnectionContext context) {
74          super(primaryImcId);
75          this.context = context;
76          try {
77              Object o = context.getAttribute(TncCommonAttributeTypeEnum
78                      .TNC_ATTRIBUTEID_MAX_MESSAGE_SIZE);
79              if (o instanceof Long) {
80                  this.maxMessageSize = ((Long) o).longValue();
81              }
82          } catch (TncException | UnsupportedOperationException e) {
83              this.maxMessageSize = HSBConstants.TCG_IM_MAX_MESSAGE_SIZE_UNKNOWN;
84          }
85      }
86  
87      @Override
88      public void sendMessage(final long messageType, final byte[] message)
89              throws TNCException {
90  
91          final int vendorIdShift = 8;
92          final int messageTypeMask = 0xFF;
93  
94          if (messageType
95                  == (TNCConstants.TNC_VENDORID_ANY << vendorIdShift
96                          | TNCConstants.TNC_SUBTYPE_ANY)) {
97  
98              throw new TNCException("Message type is set to reserved type.",
99                      TNCException.TNC_RESULT_INVALID_PARAMETER);
100         }
101 
102         if (!super.isReceiving()) {
103             throw new TNCException(
104                     "Connection is currently not allowed to receive messages.",
105                     TNCException.TNC_RESULT_ILLEGAL_OPERATION);
106         }
107 
108         this.checkMessageSize(message.length);
109 
110         long vendorId = messageType >>> vendorIdShift;
111         long type = messageType & messageTypeMask;
112 
113         TnccsMessage m = null;
114         try {
115             m = PbMessageFactoryIetf.createIm(new PbMessageImFlagEnum[0],
116                     vendorId, type, (int) super.getImId(),
117                     (int) TNCConstants.TNC_IMVID_ANY, message);
118         } catch (ValidationException e) {
119             throw new TNCException(e.getCause().getMessage(), e.getCause()
120                     .getErrorCode());
121         }
122 
123         try {
124             this.sendMessage(m);
125         } catch (TncException e) {
126             throw new TNCException(e.getMessage(), e.getResultCode().id());
127         }
128     }
129 
130     @Override
131     public void requestHandshakeRetry(final long reason) throws TNCException {
132         try {
133             this.context.requestHandshakeRetry(ImHandshakeRetryReasonEnum
134                     .fromId(reason));
135         } catch (TncException e) {
136             throw new TNCException(e.getMessage(), e.getResultCode().id());
137         }
138 
139     }
140 
141     @Override
142     public Object getAttribute(final long attributeID) throws TNCException {
143 
144         if (attributeID == TncClientAttributeTypeEnum
145                 .TNC_ATTRIBUTEID_PRIMARY_IMC_ID.id()) {
146             return new Long(super.getImId());
147         } else {
148             try {
149                 return this.context.getAttribute(DefaultTncAttributeTypeFactory
150                         .getInstance().fromId(attributeID));
151             } catch (TncException e) {
152                 throw new TNCException(e.getMessage(), e.getResultCode().id());
153             }
154         }
155     }
156 
157     @Override
158     public void setAttribute(final long attributeID,
159             final Object attributeValue) throws TNCException {
160         throw new UnsupportedOperationException(
161                 "The operation setAttribute(...) is not supported, "
162                 + "because there are no attributes to set.");
163     }
164 
165     /**
166      * Checks if message size exceeds the maximum message size, that is
167      * supported for the connection.
168      *
169      * @param size  the message size
170      * @throws TNCException if message size exceeds maximum message size
171      */
172     protected void checkMessageSize(final int size) throws TNCException {
173         if (this.maxMessageSize != HSBConstants
174                     .TCG_IM_MAX_MESSAGE_SIZE_UNKNOWN
175                 && this.maxMessageSize != HSBConstants
176                     .TCG_IM_MAX_MESSAGE_SIZE_UNLIMITED) {
177 
178             if (size > this.maxMessageSize) {
179                 throw new TNCException("Maximum message size of "
180                         + this.maxMessageSize + " exceeded.",
181                         TNCException.TNC_RESULT_EXCEEDED_MAX_MESSAGE_SIZE);
182             }
183         }
184     }
185 
186     /**
187      * Sends a message to the context for transmission.
188      *
189      * @param message the message to transmit
190      * @throws TncException if message cannot be send to the context
191      */
192     protected final void sendMessage(final TnccsMessage message)
193             throws TncException {
194         if (message != null) {
195             this.context.addMessage(message);
196         }
197     }
198 
199 }
