69  /**
70   * Generic handler to handle messages destined to/from a TNCS.
71   *
72   *
73   */
74  public class DefaultTncsHandler implements TncsHandler {
75  
76      private static final Logger LOGGER = LoggerFactory
77              .getLogger(DefaultTncsHandler.class);
78  
79      private final Attributed sessionAttributes;
80      private final String tnccLanguagePreference;
81  
82      private String tempTnccsLanguagePrefernce;
83      private TncConnectionState state;
84      private boolean handshakeStartet;
85  
86      /**
87       * Creates a TNCS message handler with the given
88       * accessible session and/or connection attributes.
89       *
90       * @param sessionAttributes the session/connection attributes
91       */
92      public DefaultTncsHandler(final Attributed sessionAttributes) {
93          NotNull.check("Attributes cannot be null. Use empty attributes instead."
94                     , sessionAttributes);
95          this.sessionAttributes = sessionAttributes;
96          this.state = DefaultTncConnectionStateEnum.HSB_CONNECTION_STATE_UNKNOWN;
97          this.tnccLanguagePreference = HSBConstants.HSB_DEFAULT_LANGUAGE;
98          this.handshakeStartet = false;
99      }
100 
101     /**
102      * Returns the accessible session and/or connection attributes.
103      *
104      * @return the sessionAttributes the session/connection attributes
105      */
106     protected Attributed getAttributes() {
107         return this.sessionAttributes;
108     }
109 
110     /**
111      * Returns the preferred language for human readable message
112      * content.
113      *
114      * @return the tnccLanguagePreference
115      */
116     protected String getTnccLanguagePreference() {
117         return this.tnccLanguagePreference;
118     }

120     @Override
121     public TncConnectionState getAccessDecision() {
122         return this.state;
123     }
124 
125     @Override
126     public void setConnectionState(final TncConnectionState state) {
127         LOGGER.debug("Connection state has changed to " + state.toString()
128                 + ".");
129         this.state = state;
130         if (this.state.equals(DefaultTncConnectionStateEnum
131                         .TNC_CONNECTION_STATE_HANDSHAKE)) {
132             this.handshakeStartet = true;
133         }
134 
135     }
136 
137     @Override
138     public List<TnccsMessage> requestMessages() {
139         List<TnccsMessage> messages = new ArrayList<>();
140         if (handshakeStartet) {
141             TnccsMessage message = null;
142             try {
143                 message = PbMessageFactoryIetf
144                         .createLanguagePreference(this.tnccLanguagePreference);
145                 messages.add(message);
146             } catch (ValidationException e) {
147                 LOGGER.warn(
148                         "TnccsMessage with language preference "
149                         + "could not be created and will be ignored.",
150                         e);
151             }
152             this.handshakeStartet = false;
153         }
154 
155         return messages;
156 
157     }
158 
159     @Override
160     public List<TnccsMessage> forwardMessage(final TnccsMessage message) {
161 
162         if (message == null || message.getValue() == null) {
163             LOGGER.debug("Because Message or message value is null, "
164                     + "it is ignored.");
165             return new ArrayList<TnccsMessage>();
166         }
167 
168         TnccsMessageValue value = message.getValue();
169 
170         LOGGER.debug("Message value received: " + value.toString());
171 
172         if (value instanceof PbMessageValueLanguagePreference) {
173             this.handleLanguagePreference(
174                     (PbMessageValueLanguagePreference) value);
175         }
176 
177         List<TnccsMessage> messages = new ArrayList<>();
178         if (handshakeStartet) {
179             TnccsMessage m = null;
180             try {
181                 m = PbMessageFactoryIetf
182                         .createLanguagePreference(this.tnccLanguagePreference);
183                 messages.add(m);
184             } catch (ValidationException e) {
185                 LOGGER.warn(
186                         "TnccsMessage with language preference could not be "
187                         + "created and will be ignored.",
188                         e);
189             }
190             this.handshakeStartet = false;
191         }
192         return messages;
193     }
194 
195     @Override
196     public void dumpMessage(final TnccsMessage message) {
197 
198         if (message == null || message.getValue() == null) {
199             LOGGER.debug("Because Message or message value is null, "
200                     + "it is ignored.");
201             return;
202         }
203 
204         TnccsMessageValue value = message.getValue();
205 
206         LOGGER.debug("Message value received: " + value.toString());
207 
208         if (value instanceof PbMessageValueLanguagePreference) {
209             this.handleLanguagePreference(
210                     (PbMessageValueLanguagePreference) value);
211         }
212     }
213 
214     /**
215      * Handles a message containing the preferred language for
216      * human readable strings of a peer.
217      *
218      * @param value the message
219      */
220     private void handleLanguagePreference(
221             final PbMessageValueLanguagePreference value) {
222 
223         String lang = value.getPreferedLanguage();
224 
225         if (lang != null && !lang.isEmpty()) {
226             this.tempTnccsLanguagePrefernce = lang.trim();
227         }
228     }
229 
230     @Override
231     public List<TnccsMessage> lastCall() {
232         if (this.tempTnccsLanguagePrefernce != null) {
233             try {
234                 this.sessionAttributes
235                         .setAttribute(TncCommonAttributeTypeEnum
236                                 .TNC_ATTRIBUTEID_PREFERRED_LANGUAGE,
237                                 this.tempTnccsLanguagePrefernce);
238 
239             } catch (TncException | UnsupportedOperationException e) {
240                 LOGGER.warn("Language preference could not be set"
241                         + " and will be ignored.");
242             }
243             this.tempTnccsLanguagePrefernce = null;
244         }
245         return new ArrayList<>();
246     }

248     @Override
249     public List<TnccsMessage> provideRecommendation(
250             final List<ImvRecommendationPair> imvResults) {
251         List<TnccsMessage> messages = new ArrayList<>(0);
252         List<ImvRecommendationPair> imvResultsCopy = imvResults;
253 
254         TncConnectionState state =
255                 DefaultTncConnectionStateEnum.TNC_CONNECTION_STATE_ACCESS_NONE;
256 
257         if (imvResultsCopy == null || imvResultsCopy.isEmpty()) {
258             imvResultsCopy = new ArrayList<>(Arrays.asList(
259                     ImvRecommendationPairFactory
260                     .getDefaultRecommendationPair()));
261         }
262         Collections.sort(imvResultsCopy,
263                 LeastPrivilegeRecommendationComparator.getInstance());
264         // because of the sort get last from list which should be the most
265         // severe //
266         ImvRecommendationPair leastEqualWeightPrivilege = imvResultsCopy
267                 .get((imvResultsCopy.size() - 1));
268 
269         messages = this
270                 .mapRecommendationToMessages(leastEqualWeightPrivilege);
271         state = this.mapRecommendationToState(leastEqualWeightPrivilege);
272 
273         this.setConnectionState(state);
274 
275         return messages;
276 
277     }

279     /**
280      * Creates messages containing the integrity check handsheck
281      * result according to the given recommendation pair.
282      *
283      * @param recommendation the overall recommendation
284      * @return a list of messages with the result
285      */
286     private List<TnccsMessage> mapRecommendationToMessages(
287             final ImvRecommendationPair recommendation) {
288 
289         List<TnccsMessage> messages = new ArrayList<>(2);
290 
291         ImvActionRecommendationEnum action = recommendation.getRecommendation();
292         ImvEvaluationResultEnum result = recommendation.getResult();
293 
294         PbMessageAccessRecommendationEnum pbAction =
295                 PbMessageAccessRecommendationEnum.DENIED;
296 
297         switch (action) {
298         case TNC_IMV_ACTION_RECOMMENDATION_ALLOW:
299             pbAction = PbMessageAccessRecommendationEnum.ALLOWED;
300             break;
301         case TNC_IMV_ACTION_RECOMMENDATION_ISOLATE:
302             pbAction = PbMessageAccessRecommendationEnum.QURANTINED;
303             break;
304         default:
305             // is already denied so do nothing
306             break;
307         }
308 
309         try {
310             TnccsMessage message = PbMessageFactoryIetf
311                     .createAccessRecommendation(pbAction);
312             messages.add(message);
313         } catch (ValidationException e) {
314             LOGGER.error(
315                     "Could not create access recommendation. "
316                     + "This should never happen.",
317                     e);
318         }
319 
320         PbMessageAssessmentResultEnum pbResult =
321                 PbMessageAssessmentResultEnum.ERROR;
322 
323         switch (result) {
324         case TNC_IMV_EVALUATION_RESULT_COMPLIANT:
325             pbResult = PbMessageAssessmentResultEnum.COMPLIANT;
326             break;
327         case TNC_IMV_EVALUATION_RESULT_NONCOMPLIANT_MAJOR:
328             pbResult = PbMessageAssessmentResultEnum.SIGNIFICANT_DIFFERENCES;
329             break;
330         case TNC_IMV_EVALUATION_RESULT_NONCOMPLIANT_MINOR:
331             pbResult = PbMessageAssessmentResultEnum.MINOR_DIFFERENCES;
332             break;
333         case TNC_IMV_EVALUATION_RESULT_DONT_KNOW:
334             pbResult = PbMessageAssessmentResultEnum.INSUFFICIANT_ATTRIBUTES;
335             break;
336         default:
337             // already an error so do nothing
338             break;
339         }
340 
341         try {
342             TnccsMessage message = PbMessageFactoryIetf
343                     .createAssessmentResult(pbResult);
344             messages.add(message);
345         } catch (ValidationException e) {
346             LOGGER.error(
347                     "Could not create assessment result. "
348                     + "This should never happen.",
349                     e);
350         }
351 
352         if (LOGGER.isDebugEnabled()) {
353             StringBuilder b = new StringBuilder();
354             b.append("TNCS recommends ")
355             .append(pbAction.toString())
356             .append(" access for the result ")
357             .append(pbResult.toString());
358 
359             try {
360                 Object connectionId = this.sessionAttributes.getAttribute(
361                         TncHsbAttributeTypeEnum.HSB_ATTRIBUTEID_IFT_ID);
362                 if (connectionId instanceof String) {
363                     b.append(" for the connection with ID ")
364                     .append((String) connectionId);
365                 }
366             } catch (TncException e) {
367                LOGGER.debug("Could not identify connection "
368                        + "via session attributes.");
369             }
370             b.append(".");
371             LOGGER.debug(b.toString());
372         }
373 
374         return messages;
375     }

377     /**
378      * Maps a action recommendation to a connection state
379      * reflecting the action.
380      *
381      * @param recommendation the overall recommendation
382      * @return a connection state
383      */
384     private TncConnectionState mapRecommendationToState(
385             final ImvRecommendationPair recommendation) {
386 
387         ImvActionRecommendationEnum action =
388                 recommendation.getRecommendation();
389         TncConnectionState state =
390                 DefaultTncConnectionStateEnum.TNC_CONNECTION_STATE_ACCESS_NONE;
391 
392         switch (action) {
393         case TNC_IMV_ACTION_RECOMMENDATION_ALLOW:
394             state = DefaultTncConnectionStateEnum
395                 .TNC_CONNECTION_STATE_ACCESS_ALLOWED;
396             break;
397         case TNC_IMV_ACTION_RECOMMENDATION_ISOLATE:
398             state = DefaultTncConnectionStateEnum
399                 .TNC_CONNECTION_STATE_ACCESS_ISOLATED;
400             break;
401         default:
402             // already none so do nothing
403             break;
404         }
405 
406         return state;
407     }
