41  import de.hsbremen.tc.tnc.message.util.ByteBuffer;

43  /**
44   * Generic builder to build a transport TNCCS batch message value compliant to
45   * RFC 6876. It can be used in a fluent way.
46   *
47   *
48   */
49  public interface PtTlsMessageValuePbBatchBuilder extends
50          TransportMessageValueBuilder {
51  
52      /**
53       * Sets the TNCC batch data.
54       *
55       * @param data the TNCCS batch data
56       * @return the builder
57       * @throws RuleException if given value is not valid
58       */
59      PtTlsMessageValuePbBatchBuilder setTnccsData(ByteBuffer data)
60              throws RuleException;
61  
62  }
