37  package org.ietf.nea.pt.value.enums;
38  
39  /**
40   * Enumeration of known transport message error codes.
41   *
42   *
43   */
44  public enum PtTlsMessageErrorCodeEnum {
45  
46      /**
47       * 0 - Reserved - Reserved value indicates that the PT-TLS Error message
48       * SHOULD be ignored by all recipients. This MAY be used for debugging
49       * purposes to allow a sender to see a copy of the message, that was
50       * received.
51       */
52      IETF_RESERVED(0),
53      /**
54       * 1 - Malformed Message - PT-TLS message unrecognized or unsupported. This
55       * error code SHOULD be sent when the basic message content sanity test
56       * fails. The sender of this error code MUST consider it a fatal error and
57       * abort the assessment.
58       */
59      IETF_MALFORMED_MESSAGE(1),
60      /**
61       * 2 - Version Not Supported - This error SHOULD be sent when a PT-TLS
62       * Responder receives a PT-TLS Version Request message containing a range of
63       * version numbers, that doesn't include any version numbers that the
64       * recipient is willing and able to support on the session. All PT-TLS
65       * messages carrying the Version Not Supported error code MUST use a version
66       * number of 1. All parties, that receive or send PT-TLS messages MUST be
67       * able to properly process an error message, that meets this description,
68       * even if they cannot process any other aspect of PT-TLS version 1. The
69       * sender and receiver of this error code MUST consider it a fatal error and
70       * close the TLS session after sending or receiving this PT-TLS message.
71       */
72      IETF_UNSUPPORTED_VERSION(2),
73      /**
74       * 3 - Type Not Supported - PT-TLS Message Type unknown or not supported.
75       * When a recipient receives a PT-TLS Message Type, that it does not support,
76       * it MUST send back this error, ignore the message, and proceed. For
77       * example, this could occur if the sender used a Vendor ID for the Message
78       * Type, that is not supported by the recipient. This error message does not
79       * indicate that a fatal error has occurred, so the assessment is allowed to
80       * continue.
81       */
82      IETF_UNSUPPORTED_MESSAGE_TYPE(3),
83      /**
84       * 4 - Invalid Message - PT-TLS message received was invalid based on the
85       * protocol state. For example, this error would be sent if a recipient
86       * receives a message associated with the PT-TLS Negotiation Phase (such as
87       * Version messages) after the protocol has reached the PT-TLS Data
88       * Transport Phase. The sender and receiver of this error code MUST consider
89       * it a fatal error and close the TLS session after sending or receiving
90       * this PT-TLS message.
91       */
92      IETF_INVALID_MESSAGE(4),
93      /**
94       * 5 - SASL Mechanism Error - A fatal error occurred while trying to perform
95       * the client authentication. For example, the NEA Client is unable to
96       * support any of the offered SASL mechanisms. The sender and receiver of
97       * this error code MUST consider it a fatal error and close the TLS session
98       * after sending or receiving this PT-TLS message.
99       */
100     IETF_SASL_MECHANISM_ERROR(5),
101     /**
102      * 6 - Invalid Parameter The PT-TLS Error message sender has received a
103      * message with an invalid or unsupported value in the PT-TLS header. This
104      * could occur if the NEA Client receives a PT-TLS message from the NEA
105      * Server with a Message Length of zero (see Section 3.5 for details). The
106      * sender and receiver of this error code MUST consider it a fatal error and
107      * close the TLS session after sending or receiving this PT-TLS message.
108      */
109     IETF_INVALID_PARAMETER(6);
110 
111     private int code;
112 
113     /**
114      * Creates an error code with the given code
115      * value.
116      * @param code the code value
117      */
118     private PtTlsMessageErrorCodeEnum(final int code) {
119         this.code = code;
120     }
121 
122     /**
123      * Returns the value of the error code.
124      * @return the error code value
125      */
126     public int code() {
127         return this.code;
128     }
129 
130     /**
131      * Returns an error code for the given code value.
132      * @param code the code value
133      * @return an error code or null
134      */
135     public static PtTlsMessageErrorCodeEnum fromCode(final int code) {
136 
137         if (code == IETF_RESERVED.code) {
138             return IETF_RESERVED;
139         }
140 
141         if (code == IETF_MALFORMED_MESSAGE.code) {
142             return IETF_MALFORMED_MESSAGE;
143         }
144 
145         if (code == IETF_UNSUPPORTED_VERSION.code) {
146             return IETF_UNSUPPORTED_VERSION;
147         }
148 
149         if (code == IETF_UNSUPPORTED_MESSAGE_TYPE.code) {
150             return IETF_UNSUPPORTED_MESSAGE_TYPE;
151         }
152 
153         if (code == IETF_INVALID_MESSAGE.code) {
154             return IETF_INVALID_MESSAGE;
155         }
156 
157         if (code == IETF_SASL_MECHANISM_ERROR.code) {
158             return IETF_SASL_MECHANISM_ERROR;
159         }
160 
161         if (code == IETF_INVALID_PARAMETER.code) {
162             return IETF_INVALID_PARAMETER;
163         }
164 
165         return null;
166     }
167 }
