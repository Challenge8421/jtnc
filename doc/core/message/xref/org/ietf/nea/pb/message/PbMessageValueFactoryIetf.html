77      /**
78       * Creates a TNCCS message value containing an integrity measurement
79       * component message. Uses the given values to address the message.
80       * @param imFlags the flags to set for the message
81       * @param subVendorId the vendor ID of the component
82       * @param subType the type of the component
83       * @param collectorId the source/destination IM(C) ID
84       * @param validatorId the source/destination IM(V) ID
85       * @param message the serialized integrity measurement component message
86       * @return an IETF RFC 5793 compliant TNCCS message value
87       */
88      public static PbMessageValueIm createImValue(
89              final PbMessageImFlagEnum[] imFlags, final long subVendorId,
90              final long subType, final int collectorId, final int validatorId,
91              final byte[] message) {
92  
93          NotNull.check("Error flags cannot be null.", (Object) imFlags);
94          byte[] imMessage = (message != null) ? message : new byte[0];
95  
96          if (subVendorId > IETFConstants.IETF_MAX_VENDOR_ID) {
97              throw new IllegalArgumentException("Vendor ID is greater than "
98                      + Long.toString(IETFConstants.IETF_MAX_VENDOR_ID) + ".");
99          }
100         if (subType > IETFConstants.IETF_MAX_TYPE) {
101             throw new IllegalArgumentException("Type is greater than "
102                     + Long.toString(IETFConstants.IETF_MAX_TYPE) + ".");
103         }
104 
105         long length = PbMessageTlvFixedLengthEnum.IM_VALUE.length()
106                 + imMessage.length;
107 
108         return new PbMessageValueIm(imFlags, subVendorId, subType, collectorId,
109                 validatorId, length, imMessage);
110     }

142     /**
143      * Creates a TNCCS message error value containing information where
144      * the error occurred in another message.
145      * @param errorFlags the flags to set for the error
146      * @param errorCode the code describing the error
147      * @param offset the offset from the message beginning to the
148      * erroneous value
149      * @return an IETF RFC 5793 compliant TNCCS message value
150      */
151     public static PbMessageValueError createErrorValueWithOffset(
152             final PbMessageErrorFlagsEnum[] errorFlags,
153             final int errorCode, final long offset) {
154         NotNull.check("Error flags cannot be null.", (Object) errorFlags);
155 
156         PbMessageValueErrorParameterOffset errorParameter =
157                 PbMessageValueErrorParameterFactoryIetf
158                 .createErrorParameterOffset(VENDORID, errorCode, offset);
159 
160         return createError(errorFlags, VENDORID, errorCode,
161                 errorParameter);
162     }

164     /**
165      * Creates a TNCCS message error value containing information about the
166      * supported version.
167      * @param errorFlags the flags to set for the error
168      * @param badVersion the unsupported version
169      * @param maxVersion the maximum supported version
170      * @param minVersion the minimum supported version
171      * @return an IETF RFC 5793 compliant TNCCS message value
172      */
173     public static PbMessageValueError createErrorValueWithVersion(
174             final PbMessageErrorFlagsEnum[] errorFlags,
175             final short badVersion, final short maxVersion,
176             final short minVersion) {
177         NotNull.check("Error flags cannot be null.", (Object) errorFlags);
178 
179         PbMessageValueErrorParameterVersion errorParameter =
180                 PbMessageValueErrorParameterFactoryIetf
181                     .createErrorParameterVersion(
182                         badVersion, maxVersion, minVersion);
183 
184         return createError(errorFlags, VENDORID,
185                 PbMessageErrorCodeEnum.IETF_UNSUPPORTED_VERSION.code(),
186                 errorParameter);
187 
188     }

190     /**
191      * Creates a TNCCS message error value containing a simple error.
192      * @param errorFlags the flags to set for the error
193      * @param errorCode the code describing the error
194      * @return an IETF RFC 5793 compliant TNCCS message value
195      */
196     public static PbMessageValueError createErrorValueSimple(
197             final PbMessageErrorFlagsEnum[] errorFlags,
198             final int errorCode) {
199         NotNull.check("Error flags cannot be null.", (Object) errorFlags);
200 
201         if (errorCode == PbMessageErrorCodeEnum.IETF_LOCAL.code()
202                     || errorCode == PbMessageErrorCodeEnum
203                         .IETF_UNEXPECTED_BATCH_TYPE.code()) {
204 
205             return createError(errorFlags, VENDORID, errorCode, null);
206         }
207 
208         throw new IllegalArgumentException(
209                 "Error parameter cannot be null for error with vendor ID "
210                         + VENDORID + " and with code " + errorCode + ".");
211     }

214      * Creates a TNCCS message value suggesting a preferred language for human
215      * readable strings.
216      * @param preferredLanguage the RFC 3282 preferred language
217      * @return an IETF RFC 5793 compliant TNCCS message value

219     public static PbMessageValueLanguagePreference
220         createLanguagePreferenceValue(
221             final String preferredLanguage) {
222

224

228         return new PbMessageValueLanguagePreference(length, lang);

245     /**
246      * Creates a TNCCS message value with a string describing a reason for an
247      * assessment result and access recommendation in the given language.
248      * @param reasonString the reason string
249      * @param langCode the RFC 4646 language code
250      * @return an IETF RFC 5793 compliant TNCCS message value
251      */
252     public static PbMessageValueReasonString createReasonStringValue(
253             final String reasonString, final String langCode) {
254 
255         String reason = (reasonString != null) ? reasonString : "";
256         String lang = (langCode != null) ? langCode : "";
257 
258         if (lang.length() > 0xFF) {
259             throw new IllegalArgumentException("Language code length "
260                     + lang.length() + "is to long.");
261         }
262 
263         long length = PbMessageTlvFixedLengthEnum.REA_STR_VALUE.length();
264         if (reason.length() > 0) {
265             length += reason.getBytes(Charset.forName("UTF-8")).length;
266         }
267         if (lang.length() > 0) {
268             length += lang.getBytes(Charset.forName("US-ASCII")).length;
269         }
270 
271         return new PbMessageValueReasonString(length, reason, lang);
272     }

278      * @param langCode the RFC 4646 language code
279      * @return an IETF RFC 5793 compliant TNCCS message value
280      */
281     public static PbMessageValueRemediationParameters
282         createRemediationParameterString(final String remediationString,
283                 final String langCode) {
284 
285         long rpVendorId = VENDORID;
286         long rpType = PbMessageRemediationParameterTypeEnum.IETF_STRING.id();
287 
288         PbMessageValueRemediationParameterString parameter =
289                 PbMessageValueRemediationParameterFactoryIetf
290                 .createRemediationParameterString(remediationString, langCode);
291 
292         return createRemediationParameter(rpVendorId, rpType, parameter);
293     }

315     /**
316      * Creates a TNCCS message value with a remediation parameter.
317      * @param rpVendorId the vendor ID of the remediation string
318      * @param rpType the type of the remediation string
319      * @param parameter the remediation parameter
320      * @return an IETF RFC 5793 compliant TNCCS message value
321      */
322     private static PbMessageValueRemediationParameters
323         createRemediationParameter(final long rpVendorId, final long rpType,
324             final AbstractPbMessageValueRemediationParameter parameter) {
325 
326         if (rpVendorId > IETFConstants.IETF_MAX_VENDOR_ID) {
327             throw new IllegalArgumentException("Vendor ID is greater than "
328                     + Long.toString(IETFConstants.IETF_MAX_VENDOR_ID) + ".");
329         }
330         if (rpType > IETFConstants.IETF_MAX_TYPE) {
331             throw new IllegalArgumentException("Type is greater than "
332                     + Long.toString(IETFConstants.IETF_MAX_TYPE) + ".");
333         }
334 
335         long length = PbMessageTlvFixedLengthEnum.REM_PAR_VALUE.length()
336                 + parameter.getLength();
337 
338         return new PbMessageValueRemediationParameters(rpVendorId, rpType,
339                 length, parameter);
340     }

342     /**
343      * Create a TNCCS message error values with an error parameter.
344      * @param flags the flags to set for the error
345      * @param errorVendorId the vendor ID of the error
346      * @param errorCode the code describing the error
347      * @param errorParameter the error parameter
348      * @return an IETF RFC 5793 compliant TNCCS message value
349      */
350     private static PbMessageValueError createError(
351             final PbMessageErrorFlagsEnum[] flags, final long errorVendorId,
352             final int errorCode,
353             final AbstractPbMessageValueErrorParameter errorParameter) {
354 
355         if (errorVendorId > IETFConstants.IETF_MAX_VENDOR_ID) {
356             throw new IllegalArgumentException("Vendor ID is greater than "
357                     + Long.toString(IETFConstants.IETF_MAX_VENDOR_ID) + ".");
358         }
359         if (errorCode > IETFConstants.IETF_MAX_ERROR_CODE) {
360             throw new IllegalArgumentException("Code is greater than "
361                     + Long.toString(IETFConstants.IETF_MAX_ERROR_CODE) + ".");
362         }
363 
364         long length = PbMessageTlvFixedLengthEnum.ERR_VALUE.length()
365                 + ((errorParameter != null) ? errorParameter.getLength() : 0);
366 
367         return new PbMessageValueError(flags, errorVendorId, errorCode, length,
368                 errorParameter);
369     }
