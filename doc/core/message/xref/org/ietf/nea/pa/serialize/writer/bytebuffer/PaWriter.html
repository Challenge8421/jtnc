37  package org.ietf.nea.pa.serialize.writer.bytebuffer;
38  
39  import java.util.HashMap;
40  import java.util.List;
41  import java.util.Map;
42  
43  import org.ietf.nea.pa.attribute.PaAttribute;
44  import org.ietf.nea.pa.attribute.PaAttributeHeader;
45  import org.ietf.nea.pa.attribute.PaAttributeValue;
46  import org.ietf.nea.pa.message.PaMessage;
47  import org.ietf.nea.pa.message.PaMessageHeader;
48  
49  import de.hsbremen.tc.tnc.message.Combined;
50  import de.hsbremen.tc.tnc.message.exception.SerializationException;
51  import de.hsbremen.tc.tnc.message.m.message.ImMessage;
52  import de.hsbremen.tc.tnc.message.m.serialize.bytebuffer.ImWriter;
53  import de.hsbremen.tc.tnc.message.util.ByteBuffer;
54  import de.hsbremen.tc.tnc.util.NotNull;
55  
56  /**
57   * Writer to serialize an entire integrity measurement component message
58   * compliant to RFC 5792 from a Java object to a buffer of bytes. The writer is
59   * composed of several writers, that are used to serialize the different message
60   * elements (e.g. header, attributes, ...). Attribute value writers can be
61   * managed with the add and remove operations.
62   *
63   *
64   */
65  public class PaWriter implements ImWriter<ImMessage>,
66          Combined<ImWriter<PaAttributeValue>> {
67  
68      private final ImWriter<PaMessageHeader> mHeadWriter;
69      private final ImWriter<PaAttributeHeader> aHeadWriter;
70      private final Map<Long, Map<Long, ImWriter<PaAttributeValue>>> valueWriters;
71  
72      /**
73       * Creates the writer with a message header writer and an attribute header
74       * writer to serialize RFC 5792 compliant headers.
75       *
76       * @param mHeadWriter the message header writer
77       * @param aHeadWriter the attribute header writer
78       */
79      PaWriter(final ImWriter<PaMessageHeader> mHeadWriter,
80              final ImWriter<PaAttributeHeader> aHeadWriter) {
81          this(mHeadWriter, aHeadWriter,
82                  new HashMap<Long, Map<Long, ImWriter<PaAttributeValue>>>());
83  
84      }
85  
86      /**
87       * Creates the writer with a message header writer, an attribute header
88       * writer and a map of attribute value writers responsible for different
89       * attribute values identified by vendor ID and type ID to serialize RFC
90       * 5792 compliant message elements.
91       *
92       * @param mHeadWriter the message header writer
93       * @param aHeadWriter the attribute header writer
94       * @param valueWriters the map of attribute value writers
95       */
96      PaWriter(final ImWriter<PaMessageHeader> mHeadWriter,
97              final ImWriter<PaAttributeHeader> aHeadWriter,
98              final Map<Long, Map<Long, ImWriter<PaAttributeValue>>>
99                  valueWriters) {
100         this.mHeadWriter = mHeadWriter;
101         this.aHeadWriter = aHeadWriter;
102         this.valueWriters = valueWriters;
103     }
104 
105     @Override
106     public void write(final ImMessage message, final ByteBuffer buffer)
107             throws SerializationException {
108         NotNull.check("Message cannot be null.", message);
109 
110         NotNull.check("Buffer cannot be null.", buffer);
111 
112         if (!buffer.isWriteable()) {
113             throw new IllegalArgumentException("Buffer must be writeable.");
114         }
115 
116         if (!(message instanceof PaMessage)) {
117             throw new IllegalArgumentException("Message of type "
118                     + message.getClass().getCanonicalName()
119                     + " is not supported. Message must be of type "
120                     + PaMessage.class.getCanonicalName() + ".");
121         }
122 
123         PaMessage paMessage = (PaMessage) message;
124 
125         /* message header */
126 
127         PaMessageHeader mHead = paMessage.getHeader();
128         mHeadWriter.write(mHead, buffer);
129 
130         /* attributes */
131         List<PaAttribute> attributes = paMessage.getAttributes();
132         if (attributes != null) {
133             for (PaAttribute paAttribute : attributes) {
134 
135                 PaAttributeHeader aHead = paAttribute.getHeader();
136                 long vendor = aHead.getVendorId();
137                 long type = aHead.getAttributeType();
138 
139                 if (valueWriters.containsKey(vendor)) {
140                     if (valueWriters.get(vendor).containsKey(type)) {
141                         aHeadWriter.write(aHead, buffer);
142                         valueWriters.get(vendor).get(type)
143                                 .write(paAttribute.getValue(), buffer);
144                     } else {
145                         throw new SerializationException(
146                                 "Attribute type is not supported.", false,
147                                 Long.toString(vendor), Long.toString(type));
148                     }
149                 } else {
150                     throw new SerializationException(
151                             "Attribute vendor is not supported.", false,
152                             Long.toString(vendor), Long.toString(type));
153                 }
154             }
155         }
156 
157     }
158 
159     @Override
160     public void add(final Long vendorId, final Long messageType,
161             final ImWriter<PaAttributeValue> writer) {
162         if (valueWriters.containsKey(vendorId)) {
163             valueWriters.get(vendorId).put(messageType, writer);
164         } else {
165             valueWriters.put(vendorId,
166                     new HashMap<Long, ImWriter<PaAttributeValue>>());
167             valueWriters.get(vendorId).put(messageType, writer);
168         }
169 
170     }
171 
172     @Override
173     public void remove(final Long vendorId, final Long messageType) {
174         if (valueWriters.containsKey(vendorId)) {
175             if (valueWriters.get(vendorId).containsKey(messageType)) {
176                 valueWriters.get(vendorId).remove(messageType);
177             }
178             if (valueWriters.get(vendorId).isEmpty()) {
179                 valueWriters.remove(vendorId);
180             }
181         }
182     }
183 
184 }
