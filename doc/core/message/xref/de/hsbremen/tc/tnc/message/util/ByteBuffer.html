37  package de.hsbremen.tc.tnc.message.util;
38  
39  /**
40   * Generic byte buffer, that can be used to write and read bytes from. It also
41   * provides several helper functions to read or write different Java primitives
42   * from/to the buffer. The buffer has two pointers to indicate the read
43   * and write position. Is is not possible to read more bytes than
44   * where written to the buffer before.
45   *
46   * The buffer can throw a underflow or overflow exception. Underflow is
47   * thrown if the bytes read exceed the bytes written. Overflow is thrown
48   * if the bytes written exceed the buffer capacity.
49   *
50   *
51   */
52  public interface ByteBuffer {
53  
54      /**
55       * Appends the content of the given buffer to
56       * the content of this buffer.
57       *
58       * @param buffer the source buffer
59       */
60      void write(ByteBuffer buffer);
61  
62      /**
63       * Appends the content of the given byte array
64       * to the content of this buffer.
65       *
66       * @param array the source array
67       */
68      void write(byte[] array);
69  
70      /**
71       * Appends the byte to the content of this buffer.
72       *
73       * @param signed the signed byte
74       */
75      void writeByte(byte signed);
76  
77      /**
78       * Appends the short as unsigned byte to the content
79       * of this buffer.
80       *
81       * @param unsigned the unsigned byte
82       */
83      void writeUnsignedByte(short unsigned);
84  
85      /**
86       * Appends the short to the content of this buffer.
87       *
88       * @param signed the signed short
89       */
90      void writeShort(short signed);
91  
92      /**
93       * Appends the int as unsigned short to the content
94       * of this buffer.
95       *
96       * @param unsigned the unsigned short
97       */
98      void writeUnsignedShort(int unsigned);
99  
100     /**
101      * Appends the int to the content of this buffer.
102      *
103      * @param signed the signed int
104      */
105     void writeInt(int signed);
106 
107     /**
108      * Appends the long as unsigned int to the content
109      * of this buffer.
110      *
111      * @param unsigned the unsigned int
112      */
113     void writeUnsignedInt(long unsigned);
114 
115     /**
116      * Appends the long to the content of this buffer.
117      *
118      * @param signed the signed long
119      */
120     void writeLong(long signed);
121 
122     /**
123      * Appends the digits of a long until length
124      * is reached. Maximum length is 8.
125      *
126      * @param number the number to append
127      * @param length the length of digits to use from the number
128      */
129     void writeDigits(long number, byte length);
130 
131     /**
132      * Returns the content of length from the buffer as
133      * byte buffer.
134      *
135      * @param length the content length to read
136      * @return the resulting buffer
137      */
138     ByteBuffer read(long length);
139 
140     /**
141      * Returns the content of length from the buffer as
142      * byte array.
143      *
144      * @param length the content length to read
145      * @return the resulting array
146      */
147     byte[] read(int length);
148 
149     /**
150      * Returns one byte from the buffer.
151      *
152      * @return the resulting byte
153      */
154     byte readByte();
155 
156     /**
157      * Returns one short from the buffer.
158      *
159      * @return the resulting short
160      */
161     short readShort();
162 
163     /**
164      * Returns one short from the buffer.
165      * The short can be build with 1 or
166      * 2 bytes depending on the given length.
167      * Maximum length is 2.
168      *
169      * Useful to read an unsigned byte.
170      *
171      * @param length the content length to read
172      * @return the resulting short
173      */
174     short readShort(byte length);
175 
176     /**
177      * Returns one int from the buffer.
178      *
179      * @return the resulting short
180      */
181     int readInt();
182 
183     /**
184      * Returns one int from the buffer.
185      * The int can be build with 1 to
186      * 4 bytes depending on the given length.
187      * Maximum length is 4.
188      *
189      * Useful to read an unsigned short.
190      *
191      * @param length the content length to read
192      * @return the resulting int
193      */
194     int readInt(byte length);
195 
196     /**
197      * Returns one long from the buffer.
198      *
199      * @return the resulting short
200      */
201     long readLong();
202 
203     /**
204      * Returns one long from the buffer.
205      * The long can be build with 1 to
206      * b bytes depending on the given length.
207      * Maximum length is 8.
208      *
209      * Useful to read an unsigned int.
210      *
211      * @param length the content length to read
212      * @return the resulting long
213      */
214     long readLong(byte length);
215 
216     /**
217      * Resets the write pointer about the given
218      * length of bytes. The new position is bytes written
219      * minus length. If the read pointer is larger than
220      * the new position, the pointer is set too, that
221      * write pointer == read pointer. If the write pointer
222      * is smaller than the given length, both pointers are
223      * set to zero. Revert write discards all bytes, that
224      * are within the reverted length.
225      *
226      * @param length the number of bytes to step back
227      * @return the new position of the write pointer
228      */
229     long revertWrite(long length);
230 
231     /**
232      * Resets the read pointer about the given
233      * length of bytes. The new position is bytes read
234      * minus length. If the read pointer is smaller than
235      * the given length, the pointer is set to zero. No bytes
236      * will be discarded by this operation.
237      *
238      * @param length the number of bytes to step back
239      * @return the new position of the write pointer
240      */
241     long revertRead(long length);
242 
243     /**
244      * Clears the buffer. The buffer will be empty
245      * after this operation an all pointers will be
246      * zero.
247      *
248      */
249     void clear();
250 
251     /**
252      * Returns the bytes currently written to
253      * the buffer. If the buffer capacity is larger
254      * than MAX_LONG and the number of bytes written
255      * is larger than MAX_LONG or the buffer is not writable
256      * -1 is returned.
257      *
258      * @return the written bytes count
259      */
260     long bytesWritten();
261 
262     /**
263      * Returns the bytes already read from
264      * the buffer. If the buffer capacity is larger
265      * than MAX_LONG and the number of bytes written
266      * is larger than MAX_LONG or the buffer is not readable
267      * -1 is returned.
268      *
269      * @return the read bytes count
270      */
271     long bytesRead();
272 
273     /**
274      * Returns true if the buffer supports reading and
275      * false if the buffer is write only (e.g. if the buffer
276      * is backed by an input stream it is read only).
277      *
278      * @return true if buffer supports reading
279      */
280     boolean isReadable();
281 
282     /**
283      * Returns true if the buffer supports writing and
284      * false if the buffer is read only (e.g. if the buffer
285      * is backed by an output stream it is write only).
286      *
287      * @return true if buffer supports writing
288      */
289     boolean isWriteable();
290 
291     /**
292      * Returns true if the buffer supports a revert of the
293      * write and read position pointers.
294      *
295      * @return true if buffer supports revert
296      */
297     boolean isRevertable();
298     /**
299      * Returns the maximum capacity of the buffer, that
300      * can be written and read afterwards. If the capacity
301      * is not known -1 is returned.
302      *
303      * @return the maximum buffer capacity
304      */
305     long capacity();
306 
307 }
