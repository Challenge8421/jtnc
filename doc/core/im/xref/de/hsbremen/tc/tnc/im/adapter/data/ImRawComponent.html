37  package de.hsbremen.tc.tnc.im.adapter.data;
38  
39  import java.util.Arrays;
40  
41  /**
42   * Holds a message concerning an integrity measurement
43   * component including the message header values and
44   * the message as raw byte array.
45   *
46   *
47   */
48  public class ImRawComponent extends AbstractImComponent {
49  
50      private final byte imFlags; // 8 bit(s)
51  
52      private final byte[] message;
53  
54      /**
55       * Creates a component with the necessary address attributes and
56       * the message as raw byte array.
57       *
58       * @param flags the message flags as composed byte
59       * @param vendorId the vendor ID describing the component
60       * @param type the type ID describing the component
61       * @param collectorId the referred IMC
62       * @param validatorId the referred IMV
63       * @param message the raw message
64       */
65      ImRawComponent(final byte flags,
66              final long vendorId, final long type,
67              final long collectorId, final long validatorId,
68              final byte[] message) {
69          super(vendorId, type, collectorId, validatorId);
70  
71          this.imFlags = flags;
72          this.message = message;
73      }
74  
75      /**
76       * Returns the flags value.
77       * @return the flags value
78       */
79      public byte getImFlags() {
80          return this.imFlags;
81      }
82  
83      /**
84       * Returns a copy of the raw message as byte array.
85       * @return the raw message
86       */
87      public byte[] getMessage() {
88          return Arrays.copyOf(message, message.length);
89      }
90  
91  }
